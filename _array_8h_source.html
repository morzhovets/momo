152 public:

156  static const bool growOnReserve = tGrowOnReserve;
157  static const bool usePtrIterator = tUsePtrIterator;
158 
159 public:
160  static size_t GrowCapacity(size_t capacity, size_t minNewCapacity,
161  ArrayGrowCause growCause, bool linear) noexcept
162  {
163  MOMO_ASSERT(capacity < minNewCapacity);
164  if (growCause == ArrayGrowCause::reserve && !growOnReserve)
165  return minNewCapacity;
166  size_t newCapacity;
167  if (capacity <= 2)
168  newCapacity = 4;
169  else if (capacity <= 64)
170  newCapacity = capacity * 2;
171  else if (linear || capacity < 150)
172  newCapacity = capacity + 64;
173  else
174  newCapacity = capacity + (capacity / 50) * 23; // k^4 < 1 + k + k^2
175  if (newCapacity < minNewCapacity)
176  newCapacity = minNewCapacity;
177  return newCapacity;
178  }

663  template<typename ItemMultiCreator>
664  void SetCountCrt(size_t count, const ItemMultiCreator& itemMultiCreator)
665  {
666  size_t newCount = count;
667  size_t initCount = GetCount();
668  size_t initCapacity = GetCapacity();
669  if (newCount <= initCount)
670  {
671  pvRemoveBack(initCount - newCount);
672  }
673  else if (newCount <= initCapacity)
674  {
675  Item* items = GetItems();
676  size_t index = initCount;
677  try
678  {
679  for (; index < newCount; ++index)
680  itemMultiCreator(items + index);
681  }
682  catch (...)
683  {
684  ItemTraits::Destroy(GetMemManager(), items + initCount, index - initCount);
685  throw;
686  }

699  itemMultiCreator(newItems + index);
