562  iterator erase(const_iterator first, const_iterator last)
563  {
564  if (first == begin() && last == end())
565  {
566  clear();
567  return end();
568  }
569  if (first == last)
570  {
571  return IteratorProxy(mHashMultiMap.MakeMutableIterator(

608  bool operator==(const unordered_multimap& right) const
609  {
610  if (mHashMultiMap.GetKeyCount() != right.mHashMultiMap.GetKeyCount())
611  return false;
612  if (mHashMultiMap.GetValueCount() != right.mHashMultiMap.GetValueCount())
613  return false;
614  for (typename HashMultiMap::ConstKeyIterator::Reference ref : mHashMultiMap.GetKeyBounds())
615  {
616  typename HashMultiMap::ConstKeyIterator keyIterRight = right.mHashMultiMap.Find(ref.key);
617  if (!keyIterRight)
618  return false;
619  if (ref.GetCount() != keyIterRight->GetCount())
620  return false;
621  if (!std::is_permutation(ref.GetBegin(), ref.GetEnd(), keyIterRight->GetBegin()))
622  return false;
623  }
624  return true;
625  }

668  template<typename... KeyArgs, typename MappedCreator>
669  iterator pvInsert(std::tuple<KeyArgs...>&& keyArgs, MappedCreator&& mappedCreator)
670  {
671  MemManager& memManager = mHashMultiMap.GetMemManager();
672  typedef momo::internal::ObjectBuffer<key_type, HashMultiMap::KeyValueTraits::keyAlignment> KeyBuffer;
673  typedef momo::internal::ObjectManager<key_type, MemManager> KeyManager;
674  typedef typename KeyManager::template Creator<KeyArgs...> KeyCreator;
675  KeyBuffer keyBuffer;
676  KeyCreator(memManager, std::move(keyArgs))(&keyBuffer);
677  iterator resIter;
678  try
679  {
680  resIter = pvInsert(std::forward_as_tuple(std::move(*&keyBuffer)),
681  std::forward<MappedCreator>(mappedCreator));
682  }
683  catch (...)
684  {
685  KeyManager::Destroy(memManager, *&keyBuffer);
686  throw;
687  }
688  KeyManager::Destroy(memManager, *&keyBuffer);
689  return resIter;
690  }
