603  InsertResult pvInsert(RKey&& key, ValueCreator&& valueCreator)
604  {
605  Position pos = Find(static_cast<const Key&>(key));
606  if (!!pos)
607  return { pos, false };
608  pos = pvAdd<false>(pos, std::forward<RKey>(key),
609  std::forward<ValueCreator>(valueCreator));

614  Position pvAdd(ConstPosition pos, RKey&& key, ValueCreator&& valueCreator)
615  {
616  auto itemCreator = [this, &key, &valueCreator] (KeyValuePair* newItem)
617  {
618  KeyValueTraits::Create(GetMemManager(), std::forward<RKey>(key),
619  std::forward<ValueCreator>(valueCreator), newItem->GetKeyPtr(),
620  newItem->GetValuePtr());
621  };

623  ConstPositionProxy::GetBasePosition(pos), std::move(itemCreator)));
