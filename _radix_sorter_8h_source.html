46  template<size_t tRadixSize = 8>
47  class RadixSorter
48  {
49  public:
50  static const size_t radixSize = tRadixSize;
51  MOMO_STATIC_ASSERT(0 < radixSize && radixSize <= 16);
52 
53  private:
54  static const size_t radixCount = size_t{1} << radixSize;
55  static const size_t selectionSortMaxCount = size_t{1} << (radixSize / 2 + 1);
56 
57  public:
58  template<typename Iterator,
59  typename CodeGetter = RadixSorterCodeGetter<Iterator>>
60  static void Sort(Iterator begin, size_t count, const CodeGetter& codeGetter = CodeGetter())
61  {
62  auto swapFunc = [] (Iterator iter1, Iterator iter2) { std::iter_swap(iter1, iter2); };
63  auto groupFunc = [] (Iterator, size_t) {};
64  Sort(begin, count, codeGetter, swapFunc, groupFunc);
65  }

67  template<typename Iterator, typename CodeGetter, typename SwapFunc, typename GroupFunc>
68  static void Sort(Iterator begin, size_t count, const CodeGetter& codeGetter,
69  const SwapFunc& swapFunc, const GroupFunc& groupFunc)
70  {
71  typedef decltype(codeGetter(begin)) Code;
72  pvSort<Code>(begin, count, codeGetter, swapFunc, groupFunc,
73  8 * sizeof(Code) - radixSize);
74  }

76  private:
77  template<typename Code, typename Iterator, typename CodeGetter,
78  typename SwapFunc, typename GroupFunc>
79  static void pvSort(Iterator begin, size_t count, const CodeGetter& codeGetter,
80  const SwapFunc& swapFunc, const GroupFunc& groupFunc, size_t shift)
81  {
82  if (count < 2)
83  return;
84  if (count == 2)
85  {
86  if (codeGetter(begin) > codeGetter(begin + 1))
87  swapFunc(begin, begin + 1);
88  return;
89  }
90  if (count <= selectionSortMaxCount)
91  pvSelectionSort<Code>(begin, count, codeGetter, swapFunc, groupFunc);
92  else
93  pvRadixSort<Code>(begin, count, codeGetter, swapFunc, groupFunc, shift);
94  }

96  template<typename Code, typename Iterator, typename CodeGetter,
97  typename SwapFunc, typename GroupFunc>
98  static void pvSelectionSort(Iterator begin, size_t count, const CodeGetter& codeGetter,
99  const SwapFunc& swapFunc, const GroupFunc& groupFunc)
100  {
101  MOMO_ASSERT(count > 0);
102  std::array<Code, selectionSortMaxCount> codes; //?
103  for (size_t i = 0; i < count; ++i)
104  codes[i] = codeGetter(UIntMath<>::Next(begin, i));
105  for (size_t i = 0; i < count - 1; ++i)
106  {
107  size_t minIndex = UIntMath<>::Dist(codes.data(),
108  std::min_element(codes.data() + i + 1, codes.data() + count));
109  if (codes[minIndex] < codes[i])
110  {
111  swapFunc(UIntMath<>::Next(begin, i), UIntMath<>::Next(begin, minIndex));
112  std::swap(codes[i], codes[minIndex]);
113  }
114  }
115  size_t prevIndex = 0;
116  for (size_t i = 1; i < count; ++i)
117  {
118  if (codes[i] != codes[prevIndex])
119  {
120  groupFunc(UIntMath<>::Next(begin, prevIndex), i - prevIndex);
121  prevIndex = i;
122  }
123  }
124  groupFunc(UIntMath<>::Next(begin, prevIndex), count - prevIndex);
125  }

127  template<typename Code, typename Iterator, typename CodeGetter,
128  typename SwapFunc, typename GroupFunc>
129  static void pvRadixSort(Iterator begin, size_t count, const CodeGetter& codeGetter,
130  const SwapFunc& swapFunc, const GroupFunc& groupFunc, size_t shift)
131  {
132  MOMO_ASSERT(count > 0);
133  std::array<size_t, radixCount> endIndexes;
134  endIndexes.fill(0);
135  Code code0 = codeGetter(begin);
136  size_t radix0 = pvGetRadix<Code>(code0, shift);
137  ++endIndexes[radix0];
138  bool singleCode = true;
139  bool singleRadix = true;
140  for (size_t i = 1; i < count; ++i)
141  {
142  Code code = codeGetter(UIntMath<>::Next(begin, i));
143  size_t radix = pvGetRadix<Code>(code, shift);
144  ++endIndexes[radix];
145  singleCode &= (code == code0);
146  singleRadix &= (radix == radix0);
147  }
148  if (singleCode)
149  return groupFunc(begin, count);
150  size_t nextShift = (shift > radixSize) ? shift - radixSize : 0;
151  if (singleRadix)
152  {
153  MOMO_ASSERT(shift > 0);
154  return pvRadixSort<Code>(begin, count, codeGetter, swapFunc, groupFunc, nextShift);
155  }
156  for (size_t r = 1; r < radixCount; ++r)
157  endIndexes[r] += endIndexes[r - 1];
158  pvRadixSort<Code>(begin, codeGetter, swapFunc, shift, endIndexes);
159  size_t beginIndex = 0;
160  if (shift > 0)
161  {
162  for (size_t e : endIndexes)
163  {
164  pvSort<Code>(UIntMath<>::Next(begin, beginIndex), e - beginIndex, codeGetter,
165  swapFunc, groupFunc, nextShift);
166  beginIndex = e;
167  }
168  }
169  else
170  {
171  for (size_t e : endIndexes)
172  {
173  groupFunc(UIntMath<>::Next(begin, beginIndex), e - beginIndex);
174  beginIndex = e;
175  }
176  }
177  }

179  template<typename Code, typename Iterator, typename CodeGetter, typename SwapFunc>
180  static void pvRadixSort(Iterator begin, const CodeGetter& codeGetter, const SwapFunc& swapFunc,
181  size_t shift, const std::array<size_t, radixCount>& endIndexes)
182  {
183  std::array<size_t, radixCount> beginIndexes;
184  beginIndexes[0] = 0;
185  for (size_t r = 1; r < radixCount; ++r)
186  beginIndexes[r] = endIndexes[r - 1];
187  for (size_t r = 0; r < radixCount; ++r)
188  {
189  size_t& beginIndex = beginIndexes[r];
190  size_t endIndex = endIndexes[r];
191  while (beginIndex < endIndex)
192  {
193  size_t radix = pvGetRadix<Code>(codeGetter(UIntMath<>::Next(begin, beginIndex)), shift);
194  if (radix != r)
195  {
196  swapFunc(UIntMath<>::Next(begin, beginIndex),
197  UIntMath<>::Next(begin, beginIndexes[radix]));
198  }
199  ++beginIndexes[radix];
200  }
201  }
202  }

206  {
207  return static_cast<size_t>(code >> shift) & ((size_t{1} << radixSize) - 1);
208  }
