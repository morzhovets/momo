385  {

392  if (memPool.pvUseCache())
393  memPool.pvFlushDeallocate();
394  mData.allocCount += memPool.mData.allocCount;
395  memPool.mData.allocCount = 0;
396  if (memPool.mFreeChunkHead == nullptr)
397  return;
398  if (mFreeChunkHead == nullptr)
399  {

401  memPool.mFreeChunkHead = nullptr;
402  return;

406  Byte* chunk = pvGetPrevChunk(memPool.mFreeChunkHead);
407  if (chunk == nullptr)

411  if (prevChunk != nullptr)
412  pvSetNextChunk(prevChunk, nextChunk);
413  pvSetPrevChunk(nextChunk, prevChunk);
414  prevChunk = pvGetPrevChunk(mFreeChunkHead);
415  nextChunk = mFreeChunkHead;
416  pvSetPrevChunk(chunk, prevChunk);
417  pvSetNextChunk(chunk, nextChunk);
418  if (prevChunk != nullptr)
419  pvSetNextChunk(prevChunk, nextChunk);
420  pvSetPrevChunk(nextChunk, prevChunk);
421  }
422  Byte* chunk = mFreeChunkHead;
423  while (true)
424  {
425  Byte* nextChunk = pvGetNextChunk(chunk);
426  if (nextChunk == nullptr)
427  break;
428  chunk = nextChunk;

599  int8_t pvGetBlockIndex(Byte* block, Byte*& chunk) const noexcept
600  {
601  const uintptr_t uipBlockSize = uintptr_t{Params::blockSize};
602  const uintptr_t uipBlockAlignment = uintptr_t{Params::blockAlignment};
603  const uintptr_t uipBlockCount = uintptr_t{Params::blockCount};
604  uintptr_t uipBlock = internal::PtrCaster::ToUInt(block);
605  MOMO_ASSERT(uipBlock % uipBlockAlignment == 0);
606  ptrdiff_t dir = static_cast<ptrdiff_t>(((uipBlock % uipBlockSize) / uipBlockAlignment) % 2);
607  ptrdiff_t index = static_cast<ptrdiff_t>((uipBlock / uipBlockSize) % uipBlockCount)
608  - (static_cast<ptrdiff_t>(uipBlockCount) & (dir - 1));
609  chunk = block - index * static_cast<ptrdiff_t>(uipBlockSize)
610  - (static_cast<ptrdiff_t>(uipBlockAlignment) & -dir);
611  return static_cast<int8_t>(index);
612  }

614  MOMO_NOINLINE Byte* pvNewChunk()
615  {
616  const uintptr_t uipBlockSize = uintptr_t{Params::blockSize};
617  const uintptr_t uipBlockAlignment = uintptr_t{Params::blockAlignment};
618  const uintptr_t uipBlockCount = uintptr_t{Params::blockCount};
619  Byte* begin = MemManagerProxy::template Allocate<Byte>(
620  GetMemManager(), pvGetChunkSize());
621  uintptr_t uipBegin = internal::PtrCaster::ToUInt(begin);
622  uintptr_t uipBlock = internal::UIntMath<uintptr_t>::Ceil(uipBegin, uipBlockAlignment);
623  uipBlock += (uipBlock % uipBlockSize) % (2 * uipBlockAlignment);
624  if ((uipBlock + uipBlockAlignment) % uipBlockSize == 0)
625  uipBlock += uipBlockAlignment;
626  if ((uipBlock / uipBlockSize) % uipBlockCount == 0)
627  uipBlock += uipBlockAlignment;
628  size_t beginOffset = static_cast<size_t>(uipBlock - uipBegin);
629  MOMO_ASSERT(beginOffset < (1 << 16));
630  Byte* block = begin + beginOffset;
631  Byte* chunk;
632  int8_t blockIndex = pvGetBlockIndex(block, chunk);
633  pvSetFirstBlockIndex(chunk, blockIndex);
634  ChunkBytes bytes;
635  bytes.firstFreeBlockIndex = blockIndex;
636  bytes.freeBlockCount = static_cast<int8_t>(uipBlockCount);
637  pvSetChunkBytes(pvGetChunkBytesPosition(chunk), bytes);
638  pvSetPrevChunk(chunk, nullptr);
639  pvSetNextChunk(chunk, nullptr);
640  pvSetBeginOffset(chunk, static_cast<uint16_t>(beginOffset));
641  for (uintptr_t i = 1; i < uipBlockCount; ++i)
642  {
643  ++blockIndex;
644  pvSetNextFreeBlockIndex(block, blockIndex);
645  block = pvGetBlock(chunk, blockIndex);
646  }
647  pvSetNextFreeBlockIndex(block, int8_t{-128});
648  return chunk;
649  }

693  template<typename BlockFilter>
694  void pvDeleteBlocks(Byte* chunk, const BlockFilter& blockFilter)
695  {
696  int8_t firstBlockIndex = pvGetFirstBlockIndex(chunk);
697  uint8_t freeBlockBits[16] = {};
698  ChunkBytes bytes = pvGetChunkBytes(pvGetChunkBytesPosition(chunk));
699  int8_t freeBlockIndex = bytes.firstFreeBlockIndex;
