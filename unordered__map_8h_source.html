847  template<typename Hint, typename... KeyArgs, typename MappedCreator>
848  std::pair<iterator, bool> pvInsert(Hint /*hint*/, std::tuple<KeyArgs...>&& keyArgs,
849  MappedCreator&& mappedCreator)
850  {
851  MemManager& memManager = mHashMap.GetMemManager();
852  typedef momo::internal::ObjectBuffer<key_type, HashMap::KeyValueTraits::keyAlignment> KeyBuffer;
853  typedef momo::internal::ObjectManager<key_type, MemManager> KeyManager;
854  typedef typename KeyManager::template Creator<KeyArgs...> KeyCreator;
855  KeyBuffer keyBuffer;
856  KeyCreator(memManager, std::move(keyArgs))(&keyBuffer);
857  bool keyDestroyed = false;
858  try
859  {
860  typename HashMap::Position pos = mHashMap.Find(*&keyBuffer);
861  if (!!pos)
862  {
863  KeyManager::Destroy(memManager, *&keyBuffer);
864  keyDestroyed = true;
865  return { IteratorProxy(pos), false };
866  }
867  auto valueCreator = [&memManager, &keyBuffer, &mappedCreator, &keyDestroyed]
868  (key_type* newKey, mapped_type* newMapped)
869  {
870  KeyManager::Relocate(memManager, *&keyBuffer, newKey);
871  keyDestroyed = true;
872  try
873  {
874  std::forward<MappedCreator>(mappedCreator)(newMapped);
875  }
876  catch (...)
877  {
878  KeyManager::Destroy(memManager, *newKey);
879  throw;
880  }
881  };
882  typename HashMap::Position resPos = mHashMap.AddCrt(pos, valueCreator);
883  return { IteratorProxy(resPos), true };
884  }
885  catch (...)
886  {
887  if (!keyDestroyed)
888  KeyManager::Destroy(memManager, *&keyBuffer);
889  throw;
890  }
891  }
