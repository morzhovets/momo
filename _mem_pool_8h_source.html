259  template<typename ResObject = void>
260  MOMO_NODISCARD ResObject* Allocate()
261  {
262  void* pblock;
263  if (Params::cachedFreeBlockCount > 0 && mCachedFreeBlocks.GetCount() > 0)
264  {

266  mCachedFreeBlocks.RemoveBack();

303  void MergeFrom(MemPool& memPool)
304  {
305  if (this == &memPool)
306  return;
307  MOMO_CHECK(MemManagerProxy::IsEqual(GetMemManager(), memPool.GetMemManager()));
308  MOMO_CHECK(static_cast<const Params&>(*this).IsEqual(memPool));
309  if (Params::cachedFreeBlockCount > 0)
310  memPool.pvFlushDeallocate();
311  mAllocCount += memPool.mAllocCount;
312  memPool.mAllocCount = 0;
313  if (memPool.mBufferHead == nullPtr)
314  return;
315  if (mBufferHead == nullPtr)
316  {
317  std::swap(mBufferHead, memPool.mBufferHead);
318  return;
319  }
320  uintptr_t buffer = mBufferHead;
321  while (true)
322  {
323  BufferPointers& pointers = pvGetBufferPointers(buffer);
324  if (pointers.nextBuffer == nullPtr)
325  break;
326  buffer = pointers.nextBuffer;
327  }
328  pvGetBufferPointers(buffer).nextBuffer = memPool.mBufferHead;
329  pvGetBufferPointers(memPool.mBufferHead).prevBuffer = buffer;
330  memPool.mBufferHead = nullPtr;
331  }

339  void pvCheckParams() const
340  {
341  MOMO_CHECK(0 < Params::blockCount && Params::blockCount < 128);
342  MOMO_CHECK(0 < Params::blockAlignment && Params::blockAlignment <= 1024);
343  MOMO_CHECK(Params::blockSize > 0);

345  MOMO_CHECK(Params::blockCount == 1 || Params::blockSize / Params::blockAlignment >= 2);
346  size_t maxBlockSize =
347  (SIZE_MAX - 2 - 3 * sizeof(void*) - 4 * Params::blockAlignment) / Params::blockCount;
348  if (Params::blockSize > maxBlockSize)

359  void pvDeleteBlock(void* pblock) noexcept
360  {
361  if (Params::blockCount > 1)
362  pvDeleteBlock(internal::PtrCaster::ToUInt(pblock));
363  else if (maxAlignment % Params::blockAlignment == 0)
364  MemManagerProxy::Deallocate(GetMemManager(), pblock, pvGetBufferSize0());
365  else
366  pvDeleteBlock1(internal::PtrCaster::ToUInt(pblock));

392  size_t bufferUsefulSize = Params::blockSize + Params::blockAlignment

403  uintptr_t pvNewBlock()
404  {
405  if (mBufferHead == nullPtr)
406  mBufferHead = pvNewBuffer();
407  uintptr_t block;
408  size_t freeBlockCount = pvNewBlock(mBufferHead, block);
409  if (freeBlockCount == 0)
410  pvRemoveBuffer(mBufferHead, false);
411  return block;
412  }

414  void pvDeleteBlock(uintptr_t block) noexcept
415  {
416  uintptr_t buffer;
417  size_t freeBlockCount = pvDeleteBlock(block, buffer);
418  if (freeBlockCount == 1)
419  {
420  BufferPointers& pointers = pvGetBufferPointers(buffer);
421  pointers.prevBuffer = nullPtr;
422  pointers.nextBuffer = mBufferHead;
423  if (mBufferHead != nullPtr)
424  pvGetBufferPointers(mBufferHead).prevBuffer = buffer;
425  mBufferHead = buffer;
426  }
427  if (freeBlockCount == Params::blockCount)
428  pvRemoveBuffer(buffer, true);
429  }
430 
431  size_t pvNewBlock(uintptr_t buffer, uintptr_t& block) noexcept
432  {
433  BufferBytes& bytes = pvGetBufferBytes(buffer);
434  block = pvGetNextFreeBlockIndex(buffer, bytes.firstFreeBlockIndex);
435  bytes.firstFreeBlockIndex = pvGetNextFreeBlockIndex(block);
436  --bytes.freeBlockCount;
437  return static_cast<size_t>(bytes.freeBlockCount);
438  }
439 
440  size_t pvDeleteBlock(uintptr_t block, uintptr_t& buffer) noexcept
441  {
442  int8_t blockIndex = pvGetBlockIndex(block, buffer);
443  BufferBytes& bytes = pvGetBufferBytes(buffer);
444  pvGetNextFreeBlockIndex(block) = bytes.firstFreeBlockIndex;
445  bytes.firstFreeBlockIndex = blockIndex;
446  ++bytes.freeBlockCount;
447  return static_cast<size_t>(bytes.freeBlockCount);
448  }

450  int8_t& pvGetNextFreeBlockIndex(uintptr_t block) noexcept
451  {
452  return *internal::PtrCaster::FromUInt<int8_t>(block);
453  }
454 
455  uintptr_t pvGetNextFreeBlockIndex(uintptr_t buffer, int8_t index) const noexcept
456  {
457  return static_cast<uintptr_t>(static_cast<intptr_t>(buffer)
458  + intptr_t{index} * static_cast<intptr_t>(Params::blockSize)
459  + (static_cast<intptr_t>(Params::blockAlignment) & -intptr_t{index >= 0}));
460  }
461 
462  int8_t pvGetBlockIndex(uintptr_t block, uintptr_t& buffer) const noexcept
463  {
464  const uintptr_t uipBlockAlignment = uintptr_t{Params::blockAlignment};
465  MOMO_ASSERT(block % uipBlockAlignment == 0);
466  uintptr_t index = (block / uintptr_t{Params::blockSize}) % uintptr_t{Params::blockCount};
467  if (((block % uintptr_t{Params::blockSize}) / uipBlockAlignment) % 2 == 1)
468  {
469  buffer = block - index * uintptr_t{Params::blockSize} - uipBlockAlignment;
470  return static_cast<int8_t>(index);
471  }
472  else
473  {
474  buffer = block + (uintptr_t{Params::blockCount} - index) * uintptr_t{Params::blockSize};
475  return static_cast<int8_t>(index) - static_cast<int8_t>(Params::blockCount);
476  }
477  }

479  uintptr_t pvNewBuffer()
480  {
481  const uintptr_t uipBlockAlignment = uintptr_t{Params::blockAlignment};
482  uintptr_t begin = internal::PtrCaster::ToUInt(
483  MemManagerProxy::Allocate(GetMemManager(), pvGetBufferSize()));
484  uintptr_t block = PMath::Ceil(begin, uipBlockAlignment);
485  block += (block % uintptr_t{Params::blockSize}) % (2 * uipBlockAlignment);
486  if ((block + uipBlockAlignment) % uintptr_t{Params::blockSize} == 0)
487  block += uipBlockAlignment;
488  if ((block / uintptr_t{Params::blockSize}) % uintptr_t{Params::blockCount} == 0)
489  block += uipBlockAlignment;
490  uintptr_t buffer;
491  int8_t blockIndex = pvGetBlockIndex(block, buffer);
492  pvGetFirstBlockIndex(buffer) = blockIndex;
493  BufferBytes& bytes = pvGetBufferBytes(buffer);
494  bytes.firstFreeBlockIndex = blockIndex;
495  bytes.freeBlockCount = static_cast<int8_t>(Params::blockCount);
496  BufferPointers& pointers = pvGetBufferPointers(buffer);
497  pointers.prevBuffer = nullPtr;
498  pointers.nextBuffer = nullPtr;
499  pointers.begin = begin;
500  for (size_t i = 1; i < Params::blockCount; ++i)
501  {
502  ++blockIndex;
503  pvGetNextFreeBlockIndex(block) = blockIndex;
504  block = pvGetNextFreeBlockIndex(buffer, blockIndex);
505  }
506  pvGetNextFreeBlockIndex(block) = int8_t{-128};
507  return buffer;
508  }
509 
510  void pvRemoveBuffer(uintptr_t buffer, bool deallocate) noexcept
511  {
512  BufferPointers& pointers = pvGetBufferPointers(buffer);
513  if (pointers.prevBuffer != nullPtr)
514  pvGetBufferPointers(pointers.prevBuffer).nextBuffer = pointers.nextBuffer;
515  if (pointers.nextBuffer != nullPtr)
516  pvGetBufferPointers(pointers.nextBuffer).prevBuffer = pointers.prevBuffer;
517  if (mBufferHead == buffer)
518  mBufferHead = pointers.nextBuffer;
519  if (deallocate)
520  {
521  MemManagerProxy::Deallocate(GetMemManager(),
522  internal::PtrCaster::FromUInt(pointers.begin), pvGetBufferSize());
523  }
524  }
525 
526  size_t pvGetBufferSize() const noexcept
527  {
528  size_t bufferUsefulSize = Params::blockCount * Params::blockSize
529  + (3 + (Params::blockSize / Params::blockAlignment) % 2) * Params::blockAlignment;
530  if ((Params::blockAlignment & (Params::blockAlignment - 1)) == 0)
531  bufferUsefulSize -= std::minmax(size_t{maxAlignment}, size_t{Params::blockAlignment}).first;
532  else
533  bufferUsefulSize -= SMath::GCD(maxAlignment, Params::blockAlignment);
534  return SMath::Ceil(bufferUsefulSize, sizeof(void*)) + 3 * sizeof(void*)
535  + ((Params::blockAlignment <= 2) ? 2 : 0);
536  }

538  int8_t& pvGetFirstBlockIndex(uintptr_t buffer) noexcept

541  }

543  BufferBytes& pvGetBufferBytes(uintptr_t buffer) noexcept
544  {
545  if (Params::blockAlignment > 2)
546  {
547  return *internal::PtrCaster::FromUInt<BufferBytes>(buffer + 1);
548  }
549  else
550  {
551  return *internal::PtrCaster::Shift<BufferBytes>(
552  &pvGetBufferPointers(buffer), sizeof(BufferPointers));

556  BufferPointers& pvGetBufferPointers(uintptr_t buffer) noexcept
557  {
558  size_t offset = Params::blockCount;
559  offset -= static_cast<size_t>(-pvGetFirstBlockIndex(buffer)); // gcc warning
560  return *internal::PtrCaster::FromUInt<BufferPointers>(
561  PMath::Ceil(buffer + uintptr_t{Params::blockAlignment + Params::blockSize * offset},
562  uintptr_t{sizeof(void*)}));
563  }

652  MOMO_ASSERT(mAllocCount > 0);
653  pvGetNextBlock(GetRealPointer(ptr)) = mBlockHead;
654  mBlockHead = ptr;
655  --mAllocCount;
656  if (mAllocCount == 0)
657  pvShrink();
658  }

666  void pvNewBuffer()
667  {
668  size_t bufferCount = mBuffers.GetCount();
669  if (bufferCount >= mMaxBufferCount)
670  throw std::length_error("momo::internal::MemPoolUInt32 length error");
671  mBuffers.Reserve(bufferCount + 1);
672  char* buffer = MemManagerProxy::template Allocate<char>(GetMemManager(),
673  pvGetBufferSize());
674  for (size_t i = 0; i < blockCount; ++i)
675  {
676  void* realPtr = buffer + mBlockSize * i;
677  pvGetNextBlock(realPtr) = (i + 1 < blockCount)
678  ? static_cast<uint32_t>(bufferCount * blockCount + i + 1) : nullPtr;
679  }
680  mBlockHead = static_cast<uint32_t>(bufferCount * blockCount);
681  mBuffers.AddBackNogrow(buffer);
682  }

684  void pvShrink() noexcept
685  {

687  {
688  pvClear();
689  mBlockHead = nullPtr;
690  mBuffers.Clear(true);
691  }
692  }

695  {

698  for (char* buffer : mBuffers)
