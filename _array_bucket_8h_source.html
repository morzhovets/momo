172  explicit ArrayBucket(Params& params, const ArrayBucket& bucket)
173  {
174  MemManager& memManager = params.GetMemManager();
175  ConstBounds bounds = bucket.GetBounds();
176  size_t count = bounds.GetCount();

178  {

183  size_t memPoolIndex = pvGetFastMemPoolIndex(count);

185  Item* items = pvGetFastItems(memory.GetPointer());
186  size_t index = 0;
187  try
188  {
189  for (; index < count; ++index)
190  ItemTraits::Copy(memManager, bounds[index], items + index);
191  }
192  catch (...)

194  ItemTraits::Destroy(memManager, items, index);
195  throw;
196  }
197  pvSet(memory.Extract(), pvMakeState(memPoolIndex, count));
198  }
199  else
200  {

202  ::new(static_cast<void*>(&pvGetArray(memory.GetPointer())))
203  Array(bounds.GetBegin(), bounds.GetEnd(), MemManagerPtr(memManager));

254  template<typename ItemCreator>
255  void AddBackCrt(Params& params, ItemCreator&& itemCreator)
256  {
257  if (mPtr == nullptr)
258  {
259  size_t newCount = 1;
260  size_t newMemPoolIndex = pvGetFastMemPoolIndex(newCount);
261  Memory memory(params.GetFastMemPool(newMemPoolIndex));
262  std::forward<ItemCreator>(itemCreator)(pvGetFastItems(memory.GetPointer()));
263  pvSet(memory.Extract(), pvMakeState(newMemPoolIndex, newCount));
264  }
265  else
266  {
267  size_t memPoolIndex = pvGetMemPoolIndex();
268  if (memPoolIndex > 0)
269  {
270  size_t count = pvGetFastCount();
271  MOMO_ASSERT(count <= memPoolIndex);
272  if (count == memPoolIndex)
273  {
274  size_t newCount = count + 1;
275  Item* items = pvGetFastItems();
276  if (newCount <= maxFastCount)
277  {
278  size_t newMemPoolIndex = pvGetFastMemPoolIndex(newCount);
279  Memory memory(params.GetFastMemPool(newMemPoolIndex));
280  Item* newItems = pvGetFastItems(memory.GetPointer());
281  ItemTraits::RelocateCreate(params.GetMemManager(), items, newItems,
282  count, std::forward<ItemCreator>(itemCreator), newItems + count);
283  params.GetFastMemPool(memPoolIndex).Deallocate(mPtr);
284  pvSet(memory.Extract(), pvMakeState(newMemPoolIndex, newCount));
285  }
286  else
287  {
288  MemPool& arrayMemPool = params.GetArrayMemPool();
289  Memory memory(arrayMemPool);
290  Array array = Array::CreateCap(maxFastCount * 2,
291  MemManagerPtr(arrayMemPool.GetMemManager()));
292  Item* newItems = array.GetItems();
293  ItemTraits::RelocateCreate(params.GetMemManager(), items, newItems,
294  count, std::forward<ItemCreator>(itemCreator), newItems + count);
295  array.SetCountCrt(newCount, [] (Item* /*newItem*/) { });
296  ::new(static_cast<void*>(&pvGetArray(memory.GetPointer())))
297  Array(std::move(array));
298  params.GetFastMemPool(memPoolIndex).Deallocate(mPtr);
299  pvSet(memory.Extract(), uint8_t{0});
300  }
301  }
302  else
303  {
304  std::forward<ItemCreator>(itemCreator)(pvGetFastItems() + count);
305  ++*mPtr;
306  }
307  }
308  else
309  {
310  pvGetArray().AddBackCrt(std::forward<ItemCreator>(itemCreator));
311  }
312  }

315  void RemoveBack(Params& params) noexcept
316  {
317  size_t count = GetBounds().GetCount();

319  if (count == 1)
