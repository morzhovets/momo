79 template<SegmentedArrayItemCountFunc tItemCountFunc = SegmentedArrayItemCountFunc::cnst,

84 template<size_t tLogInitialItemCount>
85 class SegmentedArraySettings<SegmentedArrayItemCountFunc::sqrt, tLogInitialItemCount>
86  : public internal::SegmentedArraySettingsBase
87 {
88 public:
89  static const SegmentedArrayItemCountFunc itemCountFunc = SegmentedArrayItemCountFunc::sqrt;
90  static const size_t logInitialItemCount = tLogInitialItemCount;
91 
92 public:
93  static void GetSegItemIndexes(size_t index, size_t& segIndex, size_t& itemIndex) noexcept
94  {
95  size_t index1 = (index >> logInitialItemCount) + 1;
96  size_t index2 = index & ((size_t{1} << logInitialItemCount) - 1);
97  size_t logItemCount = pvIndexToLogItemCount(index1);
98  size_t itemIndex1 = index1 & ((size_t{1} << logItemCount) - 1);
99  size_t itemIndex2 = index2;
100  segIndex = (index1 >> logItemCount) + (size_t{1} << logItemCount) - 2;
101  itemIndex = (itemIndex1 << logInitialItemCount) + itemIndex2;
102  }
103 
104  static size_t GetIndex(size_t segIndex, size_t itemIndex) noexcept
105  {
106  size_t itemIndex1 = itemIndex >> logInitialItemCount;
107  size_t itemIndex2 = itemIndex & ((size_t{1} << logInitialItemCount) - 1);
108  size_t logItemCount = pvSegIndexToLogItemCount(segIndex);
109  size_t index1 = ((segIndex + 2 - (size_t{1} << logItemCount)) << logItemCount) + itemIndex1;
110  size_t index2 = itemIndex2;
111  size_t index = ((index1 - 1) << logInitialItemCount) + index2;
112  return index;
113  }
114 
115  static size_t GetItemCount(size_t segIndex) noexcept
116  {
117  size_t logItemCount = pvSegIndexToLogItemCount(segIndex);
118  return size_t{1} << (logItemCount + logInitialItemCount);
119  }
120 
121 private:
122  static size_t pvIndexToLogItemCount(size_t index1) noexcept
123  {
124  return (internal::UIntMath<>::Log2(index1) + 1) / 2;
125  }
126 
127  static size_t pvSegIndexToLogItemCount(size_t segIndex) noexcept
128  {
129  return internal::UIntMath<>::Log2((segIndex * 2 + 4) / 3);
130  }
131 };

601  template<typename ItemMultiCreator>
602  void pvIncCount(size_t count, const ItemMultiCreator& itemMultiCreator)
603  {
604  MOMO_ASSERT(count >= mCount);
605  size_t initCapacity = GetCapacity();
606  size_t initCount = mCount;
607  if (count > initCapacity)
608  pvIncCapacity(initCapacity, count);
609  auto capacityFin = internal::Catcher::Finalize(&SegmentedArrayCore::pvDecCapacity,
610  *this, initCapacity);
611  auto countFin = internal::Catcher::Finalize(&SegmentedArrayCore::pvDecCount, *this, initCount);
612  size_t segIndex, itemIndex;
613  Settings::GetSegItemIndexes(mCount, segIndex, itemIndex);
614  while (mCount < count)
615  {
616  Item* segment = mSegments[segIndex];
617  size_t itemCount = Settings::GetItemCount(segIndex);
618  for (; itemIndex < itemCount && mCount < count; ++itemIndex, ++mCount)
619  itemMultiCreator(segment + itemIndex);
620  if (itemIndex == itemCount)
621  {
622  ++segIndex;
623  itemIndex = 0;
624  }
625  }
626  countFin.Detach();
627  capacityFin.Detach();
628  }
629 
630  void pvDecCount(size_t count) noexcept
631  {
632  MOMO_ASSERT(count <= mCount);
633  size_t segIndex, itemIndex;
634  Settings::GetSegItemIndexes(mCount, segIndex, itemIndex);
635  MemManager& memManager = GetMemManager();
636  while (mCount > count)
637  {
638  if (itemIndex == 0)
639  {
640  --segIndex;
641  itemIndex = Settings::GetItemCount(segIndex);
642  }
643  size_t remCount = internal::UIntMath<>::Min(itemIndex, mCount - count);
644  ItemTraits::Destroy(memManager, mSegments[segIndex] + itemIndex - remCount, remCount);
645  itemIndex -= remCount;
646  mCount -= remCount;
647  }
648  }
649 
650  void pvIncCapacity(size_t initCapacity, size_t capacity)
651  {
652  MOMO_ASSERT(capacity >= initCapacity);
653  size_t segIndex, itemIndex;
654  Settings::GetSegItemIndexes(capacity, segIndex, itemIndex);
655  if (itemIndex > 0)
656  ++segIndex;
657  auto fin = internal::Catcher::Finalize(&SegmentedArrayCore::pvDecCapacity, *this, initCapacity);
658  for (size_t segCount = mSegments.GetCount(); segCount < segIndex; ++segCount)
659  {
660  mSegments.Reserve(segCount + 1);
661  Item* segment = pvAllocateSegment(segCount);
662  mSegments.AddBackNogrow(segment);
663  }
664  fin.Detach();
665  }
666 
667  void pvDecCapacity(size_t capacity) noexcept
668  {
669  MOMO_ASSERT(capacity <= GetCapacity());
670  size_t segIndex, itemIndex;
671  Settings::GetSegItemIndexes(capacity, segIndex, itemIndex);
672  if (itemIndex > 0)
673  ++segIndex;
674  size_t segCount = mSegments.GetCount();
675  for (size_t i = segIndex; i < segCount; ++i)
676  pvDeallocateSegment(i, mSegments[i]);
677  mSegments.RemoveBack(segCount - segIndex);
678  }
