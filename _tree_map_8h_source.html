550  template<typename ArgIterator, typename ArgSentinel,
551  typename = decltype(internal::MapPairConverter<ArgIterator>::Convert(*std::declval<ArgIterator>()))>
552  size_t Insert(ArgIterator begin, ArgSentinel end)
553  {
554  if (begin == end)
555  return 0;
556  const TreeTraits& treeTraits = GetTreeTraits();
557  MemManager& memManager = GetMemManager();
558  size_t initCount = GetCount();
559  ArgIterator iter = std::move(begin);
560  auto pair0 = internal::MapPairConverter<ArgIterator>::Convert(*iter);
561  typedef decltype(pair0.first) KeyArg;
562  typedef decltype(pair0.second) ValueArg;
563  MOMO_STATIC_ASSERT(std::is_same<Key, typename std::decay<KeyArg>::type>::value);
564  Iterator pos = InsertVar(std::forward<KeyArg>(pair0.first),
565  std::forward<ValueArg>(pair0.second)).position;
566  for (++iter; iter != end; ++iter)
567  {
568  auto pair = internal::MapPairConverter<ArgIterator>::Convert(*iter);
569  const Key& key = pair.first;
570  if (treeTraits.IsLess(key, pos->key) || !pvIsGreater(std::next(pos), key))
571  {
572  pos = InsertVar(std::forward<KeyArg>(pair.first),
573  std::forward<ValueArg>(pair.second)).position;
574  }
575  else if (TreeTraits::multiKey || treeTraits.IsLess(pos->key, key))
576  {
577  pos = pvAdd<false>(std::next(pos), std::forward<KeyArg>(pair.first),
578  ValueCreator<ValueArg>(memManager, std::forward<ValueArg>(pair.second)));
579  }
580  }
581  return GetCount() - initCount;
582  }
