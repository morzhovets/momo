305  template<typename ItemCreator>
306  Iterator AddCrt(Params& params, ItemCreator&& itemCreator, size_t hashCode,
307  size_t logBucketCount, size_t probe)
308  {
309  Item* items = mPtrState.GetPointer();
310  size_t memPoolIndex = pvGetMemPoolIndex();
311  if (items == nullptr)
312  {
313  MOMO_ASSERT(memPoolIndex == minMemPoolIndex || memPoolIndex == maxCount);
314  pvSetHashProbe(0, hashCode, logBucketCount, probe);
315  if (memPoolIndex == minMemPoolIndex)
316  return pvAdd0<minMemPoolIndex>(params, std::forward<ItemCreator>(itemCreator), hashCode);
317  else
318  return pvAdd0<maxCount>(params, std::forward<ItemCreator>(itemCreator), hashCode);
319  }
320  else
321  {
322  size_t count = pvGetCount();
323  MOMO_ASSERT(0 < count && count <= memPoolIndex);
324  MOMO_ASSERT(count < maxCount);
325  if (count == memPoolIndex)
326  {
327  switch (memPoolIndex)
328  {
329  case 1:
330  pvSetHashProbe(1, hashCode, logBucketCount, probe);
331  return pvAdd<1>(params, std::forward<ItemCreator>(itemCreator),
332  hashCode, items);
333  case 2:
334  pvSetHashProbe(2, hashCode, logBucketCount, probe);
335  return pvAdd<2>(params, std::forward<ItemCreator>(itemCreator),
336  hashCode, items);
337  default:
338  MOMO_ASSERT(memPoolIndex == 3);
339  pvSetHashProbe(3, hashCode, logBucketCount, probe);
340  return pvAdd<3>(params, std::forward<ItemCreator>(itemCreator),
341  hashCode, items);
342  }
343  }
344  else
345  {
346  pvSetHashProbe(count, hashCode, logBucketCount, probe);
347  std::forward<ItemCreator>(itemCreator)(items + count);
348  mShortHashes[count] = pvCalcShortHash(hashCode);
349  pvSetPtrState(items, memPoolIndex);
350  return items + count;
351  }
352  }

355  template<typename ItemReplacer>
356  Iterator Remove(Params& params, Iterator iter, ItemReplacer&& itemReplacer)
357  {
358  Item* items = mPtrState.GetPointer();
359  MOMO_ASSERT(items != nullptr);
360  size_t count = pvGetCount();
361  size_t memPoolIndex = pvGetMemPoolIndex();
362  if (count == 1)
363  {
364  MOMO_ASSERT(iter == items);
365  std::forward<ItemReplacer>(itemReplacer)(*items, *items);
366  pvDeallocate<false>(params, memPoolIndex, items);
367  if (memPoolIndex != maxCount)
368  memPoolIndex = minMemPoolIndex;
369  pvSetEmpty(memPoolIndex);
370  return nullptr;
371  }
372  else
373  {
374  size_t index = UIntMath<>::Dist(items, iter);
375  MOMO_ASSERT(index < count);
376  std::forward<ItemReplacer>(itemReplacer)(items[count - 1], *iter);
377  mShortHashes[index] = mShortHashes[count - 1];
378  mShortHashes[count - 1] = emptyHashProbe;
379  if (useHashCodePartGetter && hashCount - 1 - index >= count)
380  {
381  mShortHashes[hashCount - 1 - index] = (hashCount - count >= count)
382  ? mShortHashes[hashCount - count] : emptyHashProbe;
383  }
384  pvSetPtrState(items, memPoolIndex);

386  }

389  template<typename HashCodeFullGetter>
390  size_t GetHashCodePart(const HashCodeFullGetter& hashCodeFullGetter, Iterator iter,
391  size_t bucketIndex, size_t logBucketCount, size_t newLogBucketCount)
392  {
393  if (!useHashCodePartGetter)
394  return hashCodeFullGetter();
395  Item* items = mPtrState.GetPointer();
396  size_t index = UIntMath<>::Dist(items, iter);
397  size_t hashProbe = size_t{mShortHashes[hashCount - 1 - index]};
398  bool useFullGetter = (static_cast<uint8_t>(hashProbe + 1) <= maskEmpty ||
399  (logBucketCount + logBucketCountAddend) / logBucketCountStep
400  != (newLogBucketCount + logBucketCountAddend) / logBucketCountStep);
401  if (useFullGetter)
402  return hashCodeFullGetter();
403  size_t probeShift = pvGetProbeShift(logBucketCount);
404  size_t probe = hashProbe & ((size_t{1} << probeShift) - 1);
405  size_t bucketCount = size_t{1} << logBucketCount;
406  return ((bucketIndex + bucketCount - probe) & (bucketCount - 1))
407  | (((hashProbe - size_t{maskEmpty}) >> probeShift) << logBucketCount)
408  | (size_t{mShortHashes[index]} << hashCodeShift);
409  }
