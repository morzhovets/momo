485  template<typename KeyIterator, typename ValueIterator, typename Func>
486  static void pvRelocateExec(MemManager& memManager, KeyIterator srcKeyBegin,
487  ValueIterator srcValueBegin, KeyIterator dstKeyBegin, ValueIterator dstValueBegin,
488  size_t count, Func&& func, std::false_type /*isKeyNothrowRelocatable*/,
489  std::false_type /*isValueNothrowRelocatable*/)
490  {
491  size_t keyIndex = 0;
492  size_t valueIndex = 0;
493  try
494  {
495  KeyIterator srcKeyIter = srcKeyBegin;
496  KeyIterator dstKeyIter = dstKeyBegin;
497  for (; keyIndex < count; ++keyIndex, (void)++srcKeyIter, (void)++dstKeyIter)
498  KeyManager::Copy(memManager, *srcKeyIter, std::addressof(*dstKeyIter));
499  ValueIterator srcValueIter = srcValueBegin;
500  ValueIterator dstValueIter = dstValueBegin;
501  for (; valueIndex < count; ++valueIndex, (void)++srcValueIter, (void)++dstValueIter)
502  ValueManager::Copy(memManager, *srcValueIter, std::addressof(*dstValueIter));
503  std::forward<Func>(func)();
504  }
505  catch (...)
506  {
507  for (KeyIterator itd = dstKeyBegin; keyIndex > 0; --keyIndex, (void)++itd)
508  KeyManager::Destroy(memManager, *itd);
509  for (ValueIterator itd = dstValueBegin; valueIndex > 0; --valueIndex, (void)++itd)
510  ValueManager::Destroy(memManager, *itd);
511  throw;
512  }
513  for (KeyIterator its = srcKeyBegin; keyIndex > 0; --keyIndex, (void)++its)
514  KeyManager::Destroy(memManager, *its);
515  for (ValueIterator its = srcValueBegin; valueIndex > 0; --valueIndex, (void)++its)
516  ValueManager::Destroy(memManager, *its);
517  }
