560  template<typename ArgIterator,
561  typename = decltype(internal::MapPairConverter<ArgIterator>::Convert(*std::declval<ArgIterator>()))>
562  size_t Insert(ArgIterator begin, ArgIterator end)
563  {
564  if (begin == end)
565  return 0;
566  const TreeTraits& treeTraits = GetTreeTraits();
567  MemManager& memManager = GetMemManager();
568  ArgIterator iter = begin;
569  auto pair0 = internal::MapPairConverter<ArgIterator>::Convert(*iter);
570  typedef decltype(pair0.first) KeyArg;
571  typedef decltype(pair0.second) ValueArg;
572  MOMO_STATIC_ASSERT((std::is_same<Key, typename std::decay<KeyArg>::type>::value));
573  InsertResult res = InsertVar(std::forward<KeyArg>(pair0.first),
574  std::forward<ValueArg>(pair0.second));
575  size_t count = res.inserted ? 1 : 0;
576  ++iter;
577  for (; iter != end; ++iter)
578  {
579  auto pair = internal::MapPairConverter<ArgIterator>::Convert(*iter);
580  const Key& key = pair.first;
581  const Key& prevKey = res.position->key;
582  if (treeTraits.IsLess(key, prevKey) || !pvIsGreater(std::next(res.position), key))
583  {
584  res = InsertVar(std::forward<KeyArg>(pair.first), std::forward<ValueArg>(pair.second));
585  }
586  else if (TreeTraits::multiKey || treeTraits.IsLess(prevKey, key))
587  {
588  res.position = pvAdd<false>(std::next(res.position), std::forward<KeyArg>(pair.first),
589  ValueCreator<ValueArg>(memManager, std::forward<ValueArg>(pair.second)));
590  res.inserted = true;
591  }
592  else
593  {
594  res.inserted = false;
595  }
596  count += res.inserted ? 1 : 0;
597  }
598  return count;
599  }
