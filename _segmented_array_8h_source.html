68 template<size_t tLogInitialItemCount>
69 class SegmentedArraySettings<SegmentedArrayItemCountFunc::sqrt, tLogInitialItemCount>
70 {
71 public:
72  static const CheckMode checkMode = CheckMode::bydefault;
73 
74  static const SegmentedArrayItemCountFunc itemCountFunc = SegmentedArrayItemCountFunc::sqrt;
75  static const size_t logInitialItemCount = tLogInitialItemCount;
76 
77  typedef ArraySettings<> SegmentsSettings;
78 
79 public:
80  static void GetSegItemIndexes(size_t index, size_t& segIndex, size_t& itemIndex) noexcept
81  {
82  size_t index1 = (index >> logInitialItemCount) + 1;
83  size_t index2 = index & ((size_t{1} << logInitialItemCount) - 1);
84  size_t logItemCount = pvIndexToLogItemCount(index1);
85  size_t itemIndex1 = index1 & ((size_t{1} << logItemCount) - 1);
86  size_t itemIndex2 = index2;
87  segIndex = (index1 >> logItemCount) + (size_t{1} << logItemCount) - 2;
88  itemIndex = (itemIndex1 << logInitialItemCount) + itemIndex2;
89  }
90 
91  static size_t GetIndex(size_t segIndex, size_t itemIndex) noexcept
92  {
93  size_t itemIndex1 = itemIndex >> logInitialItemCount;
94  size_t itemIndex2 = itemIndex & ((size_t{1} << logInitialItemCount) - 1);
95  size_t logItemCount = pvSegIndexToLogItemCount(segIndex);
96  size_t index1 = ((segIndex + 2 - (size_t{1} << logItemCount)) << logItemCount) + itemIndex1;
97  size_t index2 = itemIndex2;
98  size_t index = ((index1 - 1) << logInitialItemCount) + index2;
99  return index;
100  }
101 
102  static size_t GetItemCount(size_t segIndex) noexcept
103  {
104  size_t logItemCount = pvSegIndexToLogItemCount(segIndex);
105  return size_t{1} << (logItemCount + logInitialItemCount);
106  }
107 
108 private:
109  static size_t pvIndexToLogItemCount(size_t index1) noexcept
110  {
111  return (internal::UIntMath<>::Log2(index1) + 1) / 2;
112  }
113 
114  static size_t pvSegIndexToLogItemCount(size_t segIndex) noexcept
115  {
116  return internal::UIntMath<>::Log2((segIndex * 2 + 4) / 3);
117  }
118 };

632  template<typename ItemMultiCreator>
633  void pvIncCount(size_t count, const ItemMultiCreator& itemMultiCreator)
634  {
635  MOMO_ASSERT(count >= mCount);
636  size_t initCapacity = GetCapacity();
637  size_t initCount = mCount;
638  if (count > initCapacity)
639  pvIncCapacity(initCapacity, count);
640  try
641  {
642  size_t segIndex, itemIndex;
643  Settings::GetSegItemIndexes(mCount, segIndex, itemIndex);
644  while (mCount < count)
645  {
646  Item* segment = mSegments[segIndex];
647  size_t itemCount = Settings::GetItemCount(segIndex);
648  for (; itemIndex < itemCount && mCount < count; ++itemIndex, ++mCount)
649  itemMultiCreator(segment + itemIndex);
650  if (itemIndex == itemCount)
651  {
652  ++segIndex;
653  itemIndex = 0;
654  }
655  }
656  }
657  catch (...)
658  {
659  pvDecCount(initCount);
660  pvDecCapacity(initCapacity);
661  throw;
662  }
663  }

665  void pvDecCount(size_t count) noexcept
666  {
667  MOMO_ASSERT(count <= mCount);
668  size_t segIndex, itemIndex;
669  Settings::GetSegItemIndexes(mCount, segIndex, itemIndex);
670  MemManager& memManager = GetMemManager();
671  while (mCount > count)
672  {
673  if (itemIndex == 0)
674  {
675  --segIndex;
676  itemIndex = Settings::GetItemCount(segIndex);
677  }
678  size_t remCount = std::minmax(itemIndex, mCount - count).first;
679  ItemTraits::Destroy(memManager, mSegments[segIndex] + itemIndex - remCount, remCount);
680  itemIndex -= remCount;
681  mCount -= remCount;
682  }
683  }

685  void pvIncCapacity(size_t initCapacity, size_t capacity)
686  {
687  MOMO_ASSERT(capacity >= initCapacity);
688  size_t segIndex, itemIndex;
689  Settings::GetSegItemIndexes(capacity, segIndex, itemIndex);
690  if (itemIndex > 0)
691  ++segIndex;
692  try
693  {
694  for (size_t segCount = mSegments.GetCount(); segCount < segIndex; ++segCount)
695  {
696  mSegments.Reserve(segCount + 1);
697  Item* segment = pvAllocateSegment(segCount);
698  mSegments.AddBackNogrow(segment);
699  }
700  }
701  catch (...)
702  {
703  pvDecCapacity(initCapacity);
704  throw;
705  }
706  }

708  void pvDecCapacity(size_t capacity) noexcept
709  {
710  MOMO_ASSERT(capacity <= GetCapacity());
711  size_t segIndex, itemIndex;
712  Settings::GetSegItemIndexes(capacity, segIndex, itemIndex);
713  if (itemIndex > 0)
714  ++segIndex;
715  size_t segCount = mSegments.GetCount();
716  for (size_t i = segIndex; i < segCount; ++i)
717  pvDeallocateSegment(i, mSegments[i]);
718  mSegments.RemoveBack(segCount - segIndex);
719  }
