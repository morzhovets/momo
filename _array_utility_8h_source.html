196  static void InsertNogrow(Array& array, size_t index, size_t count, const Item& item)
197  {
198  size_t initCount = array.GetCount();
199  MOMO_CHECK(index <= initCount);
200  MOMO_ASSERT(array.GetCapacity() >= initCount + count);
201  if (count == 0)
202  return;
203  MemManager& memManager = array.GetMemManager();
204  if (index + count < initCount)
205  {
206  for (size_t i = initCount - count; i < initCount; ++i)
207  array.AddBackNogrow(std::move(array[i]));
208  for (size_t i = initCount - count; i > index; --i)
209  ItemTraits::Assign(memManager, std::move(array[i - 1]), array[i + count - 1]);
210  for (size_t i = index; i < index + count; ++i)
211  ItemTraits::Assign(memManager, item, array[i]);
212  }
213  else
214  {
215  for (size_t i = initCount; i < index + count; ++i)
216  array.AddBackNogrow(item);
217  for (size_t i = index; i < initCount; ++i)
218  {
219  Item& arrayItem = array[i];

277  static void Remove(Array& array, size_t index, size_t count)
278  {
279  size_t initCount = array.GetCount();
280  MOMO_CHECK(index + count <= initCount);
281  if (count == 0)
282  return;
283  MemManager& memManager = array.GetMemManager();
284  for (size_t i = index + count; i < initCount; ++i)
285  ItemTraits::Assign(memManager, std::move(array[i]), array[i - count]);
286  array.RemoveBack(count);

289  template<typename ItemFilter>
290  static size_t Remove(Array& array, const ItemFilter& itemFilter)
291  {
292  size_t initCount = array.GetCount();
293  size_t newCount = 0;
294  while (newCount < initCount && !itemFilter(static_cast<const Item&>(array[newCount])))
295  ++newCount;
296  MemManager& memManager = array.GetMemManager();
297  for (size_t i = newCount + 1; i < initCount; ++i)
298  {
299  if (itemFilter(static_cast<const Item&>(array[i])))
300  continue;
301  ItemTraits::Assign(memManager, std::move(array[i]), array[newCount]);
302  ++newCount;
303  }
304  size_t remCount = initCount - newCount;
305  array.RemoveBack(remCount);
306  return remCount;
