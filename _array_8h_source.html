135 template<size_t tInternalCapacity = 0,
136  bool tGrowOnReserve = true,
137  bool tUsePtrIterator = (tInternalCapacity == 0)>
138 class ArraySettings
139 {
140 public:
141  static const CheckMode checkMode = CheckMode::bydefault;
142 
143  static const size_t internalCapacity = tInternalCapacity;
144  static const bool growOnReserve = tGrowOnReserve;
145  static const bool usePtrIterator = tUsePtrIterator;
146 
147 public:
148  static size_t GrowCapacity(size_t capacity, size_t minNewCapacity,
149  ArrayGrowCause growCause, bool realloc)
150  {
151  MOMO_ASSERT(capacity < minNewCapacity);
152  if (growCause == ArrayGrowCause::reserve && !growOnReserve)
153  return minNewCapacity;
154  size_t newCapacity;
155  if (capacity <= 2)
156  newCapacity = 4;
157  else if (capacity < 256)
158  newCapacity = capacity * 2;
159  else if (capacity < 8192)
160  newCapacity = capacity + capacity / 2;
161  else if (realloc && capacity <= SIZE_MAX - 4096)
162  newCapacity = capacity + 4096;
163  else if (!realloc && capacity <= (SIZE_MAX / 146) * 100 + 99)
164  newCapacity = (capacity / 100) * 146; // k^4 < 1 + k + k^2
165  else
166  throw std::length_error("momo::ArraySettings length error");
167  if (newCapacity < minNewCapacity)
168  newCapacity = minNewCapacity;
169  return newCapacity;
170  }
171 };

615  template<typename MultiItemCreator>
616  void SetCountCrt(size_t count, const MultiItemCreator& multiItemCreator)
617  {
618  size_t newCount = count;
619  size_t initCount = GetCount();
620  size_t initCapacity = GetCapacity();
621  if (newCount <= initCount)
622  {
623  pvRemoveBack(initCount - newCount);
624  }
625  else if (newCount <= initCapacity)
626  {
627  Item* items = GetItems();
628  size_t index = initCount;
629  try
630  {
631  for (; index < newCount; ++index)
632  multiItemCreator(items + index);
633  }
634  catch (...)
635  {
636  ItemTraits::Destroy(GetMemManager(), items + initCount, index - initCount);
637  throw;
638  }
639  mData.SetCount(newCount);
640  }
641  else
642  {
643  size_t newCapacity = pvGrowCapacity(initCapacity, newCount,
644  ArrayGrowCause::reserve, false);
645  auto itemsRelocator = [this, initCount, newCount, &multiItemCreator] (Item* newItems)
646  {
647  size_t index = initCount;
648  try
649  {
650  for (; index < newCount; ++index)
651  multiItemCreator(newItems + index);
652  ItemTraits::Relocate(GetMemManager(), GetItems(), newItems, initCount);
653  }
654  catch (...)
655  {
656  ItemTraits::Destroy(GetMemManager(), newItems + initCount, index - initCount);
657  throw;
658  }
659  };
660  mData.Reset(newCapacity, newCount, itemsRelocator);
661  }
662  }
