708  std::pair<iterator, bool> pvFind(const_iterator hint, const key_type& key)
709  {
710  if (hint != begin() && !pvIsOrdered(std::prev(hint)->first, key))
711  return pvFind(nullptr, key);
712  if (hint != end() && !pvIsOrdered(key, hint->first))
