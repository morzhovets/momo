152 template<size_t tInternalCapacity = 0,
153  bool tGrowOnReserve = true,
154  bool tUsePtrIterator = (tInternalCapacity == 0)>
155 class ArraySettings
156 {
157 public:
158  static const CheckMode checkMode = CheckMode::bydefault;
159  static const bool allowExceptionSuppression = true;
160 
161  static const size_t internalCapacity = tInternalCapacity;
162  static const bool growOnReserve = tGrowOnReserve;
163  static const bool usePtrIterator = tUsePtrIterator;
164 
165 public:
166  static size_t GrowCapacity(size_t capacity, size_t minNewCapacity,
167  ArrayGrowCause growCause, bool linear) noexcept
168  {
169  MOMO_ASSERT(capacity < minNewCapacity);
170  if (growCause == ArrayGrowCause::reserve && !growOnReserve)
171  return minNewCapacity;
172  size_t newCapacity;
173  if (capacity <= 2)
174  newCapacity = 4;
175  else if (capacity <= 64)
176  newCapacity = capacity * 2;
177  else if (linear || capacity < 150)
178  newCapacity = capacity + 64;
179  else
180  newCapacity = capacity + (capacity / 50) * 23; // k^4 < 1 + k + k^2
181  if (newCapacity < minNewCapacity)
182  newCapacity = minNewCapacity;
183  return newCapacity;
184  }
185 };

670  template<typename ItemMultiCreator>
671  void SetCountCrt(size_t count, const ItemMultiCreator& itemMultiCreator)
672  {
673  size_t newCount = count;
674  size_t initCount = GetCount();
675  size_t initCapacity = GetCapacity();
676  if (newCount <= initCount)
677  {
678  pvRemoveBack(initCount - newCount);
679  }
680  else if (newCount <= initCapacity)
681  {
682  Item* items = GetItems();
683  size_t index = initCount;
684  auto fin = internal::Catcher::Finalize(&ArrayCore::pvDestroyExtraItems,
685  *this, items, initCount, index);
686  for (; index < newCount; ++index)
687  itemMultiCreator(items + index);
688  fin.Detach();

690  }

694  ArrayGrowCause::reserve, false);

699  *this, newItems, initCount, index);

703  fin.Detach();
