66 class SegmentedArraySettings<SegmentedArrayItemCountFunc::sqrt, tLogInitialItemCount>
67 {
68 public:
69  static const CheckMode checkMode = CheckMode::bydefault;
70 
71  static const SegmentedArrayItemCountFunc itemCountFunc = SegmentedArrayItemCountFunc::sqrt;
72  static const size_t logInitialItemCount = tLogInitialItemCount;
73 
74  typedef ArraySettings<> SegmentsSettings;
75 
76 public:
77  static void GetSegItemIndexes(size_t index, size_t& segIndex, size_t& itemIndex) noexcept
78  {
79  size_t index1 = (index >> logInitialItemCount) + 1;
80  size_t index2 = index & ((size_t{1} << logInitialItemCount) - 1);
81  size_t logItemCount = pvIndexToLogItemCount(index1);
82  size_t itemIndex1 = index1 & ((size_t{1} << logItemCount) - 1);
83  size_t itemIndex2 = index2;
84  segIndex = (index1 >> logItemCount) + (size_t{1} << logItemCount) - 2;
85  itemIndex = (itemIndex1 << logInitialItemCount) + itemIndex2;
86  }
87 
88  static size_t GetIndex(size_t segIndex, size_t itemIndex) noexcept
89  {
90  size_t itemIndex1 = itemIndex >> logInitialItemCount;
91  size_t itemIndex2 = itemIndex & ((size_t{1} << logInitialItemCount) - 1);
92  size_t logItemCount = pvSegIndexToLogItemCount(segIndex);
93  size_t index1 = ((segIndex + 2 - (size_t{1} << logItemCount)) << logItemCount) + itemIndex1;
94  size_t index2 = itemIndex2;
95  size_t index = ((index1 - 1) << logInitialItemCount) + index2;
96  return index;
97  }
98 
99  static size_t GetItemCount(size_t segIndex) noexcept
100  {
101  size_t logItemCount = pvSegIndexToLogItemCount(segIndex);
102  return size_t{1} << (logItemCount + logInitialItemCount);
103  }
104 
105 private:
106  static size_t pvIndexToLogItemCount(size_t index1) noexcept
107  {
108  return (internal::UIntMath<>::Log2(index1) + 1) / 2;
109  }
110 
111  static size_t pvSegIndexToLogItemCount(size_t segIndex) noexcept
112  {
113  return internal::UIntMath<>::Log2((segIndex * 2 + 4) / 3);

622  template<typename MultiItemCreator>
623  void pvIncCount(size_t count, const MultiItemCreator& multiItemCreator)
624  {
625  MOMO_ASSERT(count >= mCount);
626  size_t initCapacity = GetCapacity();
627  size_t initCount = mCount;
628  if (count > initCapacity)
629  pvIncCapacity(initCapacity, count);
630  try
631  {
632  size_t segIndex, itemIndex;
633  Settings::GetSegItemIndexes(mCount, segIndex, itemIndex);
634  while (mCount < count)
635  {
636  Item* segment = mSegments[segIndex];
637  size_t itemCount = Settings::GetItemCount(segIndex);
638  for (; itemIndex < itemCount && mCount < count; ++itemIndex, ++mCount)
639  multiItemCreator(segment + itemIndex);
640  if (itemIndex == itemCount)
641  {
642  ++segIndex;
643  itemIndex = 0;
644  }
645  }
646  }
647  catch (...)
648  {
649  pvDecCount(initCount);
650  pvDecCapacity(initCapacity);
651  throw;
652  }
653  }

655  void pvDecCount(size_t count) noexcept
656  {
657  MOMO_ASSERT(count <= mCount);
658  size_t segIndex, itemIndex;
659  Settings::GetSegItemIndexes(mCount, segIndex, itemIndex);
660  MemManager& memManager = GetMemManager();
661  while (mCount > count)
662  {
663  if (itemIndex == 0)
664  {
665  --segIndex;
666  itemIndex = Settings::GetItemCount(segIndex);
667  }
668  size_t delCount = std::minmax(itemIndex, mCount - count).first;
669  ItemTraits::Destroy(memManager, mSegments[segIndex] + itemIndex - delCount, delCount);
670  itemIndex -= delCount;
671  mCount -= delCount;
672  }
673  }

676  {
677  MOMO_ASSERT(capacity >= initCapacity);
678  size_t segIndex, itemIndex;
679  Settings::GetSegItemIndexes(capacity, segIndex, itemIndex);
680  if (itemIndex > 0)
681  ++segIndex;
682  try

685  {
686  mSegments.Reserve(segCount + 1);
687  Item* segMemory = pvGetSegMemory(segCount);
688  mSegments.AddBackNogrow(segMemory);
689  }
