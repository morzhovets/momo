188  static void Insert(Array& array, size_t index, size_t count, const Item& item)
189  {
190  size_t initCount = array.GetCount();
191  MOMO_CHECK(index <= initCount);
192  MOMO_ASSERT(array.GetCapacity() >= initCount + count);
193  MemManager& memManager = array.GetMemManager();
194  if (index + count < initCount)
195  {
196  for (size_t i = initCount - count; i < initCount; ++i)
197  array.AddBackNogrow(std::move(array[i]));
198  for (size_t i = initCount - count; i > index; --i)
199  ItemTraits::Assign(memManager, std::move(array[i - 1]), array[i + count - 1]);
200  for (size_t i = index; i < index + count; ++i)
201  ItemTraits::Assign(memManager, item, array[i]);
202  }
203  else
204  {
205  for (size_t i = initCount; i < index + count; ++i)
206  array.AddBackNogrow(item);
207  for (size_t i = index; i < initCount; ++i)
208  {
209  Item& arrayItem = array[i];
210  array.AddBackNogrow(std::move(arrayItem));
211  ItemTraits::Assign(memManager, item, arrayItem);
212  }

275  static size_t Remove(Array& array, const Predicate& pred)
276  {
277  size_t initCount = array.GetCount();
278  size_t newCount = 0;
279  while (newCount < initCount && !pred(static_cast<const Item&>(array[newCount])))
280  ++newCount;
281  MemManager& memManager = array.GetMemManager();
282  for (size_t i = newCount + 1; i < initCount; ++i)
283  {
284  if (pred(static_cast<const Item&>(array[i])))
285  continue;
286  ItemTraits::Assign(memManager, std::move(array[i]), array[newCount]);
287  ++newCount;
288  }
289  size_t remCount = initCount - newCount;
290  array.RemoveBack(remCount);
