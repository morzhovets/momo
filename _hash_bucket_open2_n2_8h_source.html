166  template<typename ItemReplacer>
167  Iterator Remove(Params& /*params*/, Iterator iter, ItemReplacer&& itemReplacer)
168  {
169  size_t count = pvGetCount();

171  MOMO_ASSERT(index >= maxCount - count);
172  std::forward<ItemReplacer>(itemReplacer)(*&mItems[maxCount - count], *&mItems[index]);
173  mHashData.shortHashes[index] = mHashData.shortHashes[maxCount - count];
174  mHashData.shortHashes[maxCount - count] = emptyShortHash;
175  if (useHashCodePartGetter)

181  template<typename HashCodeFullGetter>
182  size_t GetHashCodePart(const HashCodeFullGetter& hashCodeFullGetter, Iterator iter,
183  size_t bucketIndex, size_t logBucketCount, size_t newLogBucketCount)
184  {
185  if (!useHashCodePartGetter)
186  return hashCodeFullGetter();
187  size_t index = UIntMath<>::Dist(&mItems[0], std::addressof(*iter));
188  uint8_t hashProbe = mHashData.hashProbes[index];
189  bool useFullGetter = (hashProbe == emptyHashProbe ||
190  (logBucketCount + logBucketCountAddend) / logBucketCountStep
191  != (newLogBucketCount + logBucketCountAddend) / logBucketCountStep);
192  if (useFullGetter)
193  return hashCodeFullGetter();
194  size_t probeShift = pvGetProbeShift(logBucketCount);
195  MOMO_ASSERT(probeShift > 0);
196  size_t probe = size_t{hashProbe} & ((size_t{1} << probeShift) - 1);
197  size_t probe2 = (probe % 2 == 0) ? (probe / 2) * (probe + 1) : probe * ((probe + 1) / 2);
198  size_t bucketCount = size_t{1} << logBucketCount;
199  return ((bucketIndex - probe2) & (bucketCount - 1))
200  | ((size_t{hashProbe} >> probeShift) << logBucketCount)
201  | (size_t{mHashData.shortHashes[index]} << hashCodeShift);
202  }
